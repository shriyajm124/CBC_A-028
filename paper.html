let currentQuestion = 0;

let answers = [];


function renderQuestion() {

    const container = document.getElementById("questionContainer");

    const q = quiz[currentQuestion];

    let html = `<h2>${q.question}</h2><div class="options">`;

    q.options.forEach((opt, i) => {

        const checked = answers[currentQuestion] === i ? 'checked' : '';

        html += `

            <label>

                <input type="radio" name="option" value="${i}" ${checked}>

                ${opt.text}

            </label>

        `;

    });

    html += '</div>';

    container.innerHTML = html;


    document.getElementById("prevBtn").disabled = currentQuestion === 0;

    document.getElementById("nextBtn").innerText = currentQuestion === quiz.length - 1 ? "Submit" : "Next";

}


function nextQuestion() {

    const selected = document.querySelector('input[name="option"]:checked');

    if (!selected) {

        alert("Please select an option.");

        return;

    }

    answers[currentQuestion] = parseInt(selected.value);

    if (currentQuestion < quiz.length - 1) {

        currentQuestion++;

        renderQuestion();

    } else {

        showResults();

    }

}


function prevQuestion() {

    if (currentQuestion > 0) {

        currentQuestion--;

        renderQuestion();

    }

}


function showResults() {

    const tags = {};

    answers.forEach((ansIndex, i) => {

        quiz[i].options[ansIndex].tags.forEach(tag => {

            tags[tag] = (tags[tag] || 0) + 1;

        });

    });


    const result = destinations.filter(dest => {

        return dest.tags.some(tag => tags[tag]);

    });


    const resultsContainer = document.getElementById("results");

    let html = "<h2>Your Travel Matches</h2>";

    result.forEach(dest => {

        html += `

            <div class="destination">

                <h3>${dest.name}</h3>

                <p>${dest.description}</p>

                <div class="eco-stays">

                    ${dest.stays.map(stay => `<a href="${stay.url}" target="_blank">${stay.name}</a>`).join("<br>")}

                </div>

            </div>

        `;

    });

    document.getElementById("quizForm").style.display = "none";

    resultsContainer.innerHTML = html;

}
